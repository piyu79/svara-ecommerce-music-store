<script>
  function applyFilters() {
    const priceVal = document.getElementById('guitar-price').value;
    const searchVal = document.getElementById('guitar-search').value.toLowerCase().trim();
    const brandVal = document.getElementById('guitar-brand').value;
    const sortVal = document.getElementById('guitar-sort').value;
    const grid = document.getElementById('productsGrid');
    const cards = Array.from(document.querySelectorAll('#productsGrid .product-card'));
    let visibleCount = 0;

    cards.forEach(card => {
      const price = parseFloat(card.getAttribute('data-price'));
      const brand = card.getAttribute('data-brand');
      const title = card.querySelector('h3').textContent.toLowerCase();
      const description = card.querySelector('.description').textContent.toLowerCase();

      let priceMatch = true;
      if (priceVal === 'Under ₹1000') priceMatch = price < 1000;
      else if (priceVal === '₹1000 - ₹2500') priceMatch = price >= 1000 && price <= 2500;
      else if (priceVal === '₹2500+') priceMatch = price > 2500;

      const brandMatch = (brandVal === 'All Brands') || (brand === brandVal);
      const searchMatch = !searchVal || title.includes(searchVal) || description.includes(searchVal) || brand.toLowerCase().includes(searchVal);

      const show = priceMatch && brandMatch && searchMatch;
      card.style.display = show ? '' : 'none';
      if (show) visibleCount++;
    });

    document.getElementById('noResultsMsg').style.display = visibleCount === 0 ? 'block' : 'none';
    updateProductCount(visibleCount);

    cards.sort((a, b) => {
      const priceA = parseFloat(a.getAttribute('data-price'));
      const priceB = parseFloat(b.getAttribute('data-price'));
      const nameA = a.querySelector('h3').textContent.toLowerCase();
      const nameB = b.querySelector('h3').textContent.toLowerCase();

      if (sortVal === 'Price: Low to High') return priceA - priceB;
      if (sortVal === 'Price: High to Low') return priceB - priceA;
      if (sortVal === 'Name: A to Z') return nameA.localeCompare(nameB);
      return 0;
    });

    cards.forEach(card => grid.appendChild(card));
  }

  function updateProductCount(count) {
    const noun = count === 1 ? 'guitar' : 'guitars';
    document.getElementById('productCount').textContent = 'Showing ' + count + ' ' + noun;
  }

  document.getElementById('applyFilters').addEventListener('click', applyFilters);
  document.getElementById('guitar-search').addEventListener('input', applyFilters);
  document.getElementById('guitar-price').addEventListener('change', applyFilters);
  document.getElementById('guitar-brand').addEventListener('change', applyFilters);
  document.getElementById('guitar-sort').addEventListener('change', applyFilters);
  applyFilters();
</script>

<script>
  function addToCart(product) {
    const userId = localStorage.getItem('svara_user_id');
    if (!userId) {
      showToast('Please log in to add items to your cart', true);
      return Promise.resolve(false);
    }

    const formData = new FormData();
    formData.append('user_id', userId);
    formData.append('product_name', product.name);
    formData.append('product_price', product.price);
    formData.append('product_image', product.image);

    return fetch('cart_add.php', { method: 'POST', body: formData })
      .then(res => res.json())
      .then(data => {
        if (!data.success) {
          showToast(data.message || 'Could not add item to cart', true);
        }
        return data.success;
      })
      .catch(() => {
        showToast('Network error - please try again', true);
        return false;
      });
  }
 
  function showToast(message, isError) {
    const toast = document.getElementById('toast');
    document.getElementById('toast-msg').textContent = message;
    toast.style.background = isError ? '#96281B' : '';
    toast.classList.add('show');
    clearTimeout(window._toastTimer);
    window._toastTimer = setTimeout(() => toast.classList.remove('show'), 3000);
  }
 
  document.querySelectorAll('#productsGrid .buy-button').forEach(button => {

      const card = this.closest('.product-card');
      const productName = card.querySelector('h3').textContent;
      const productPrice = card.querySelector('.price').textContent;
      const productImage = card.querySelector('img').getAttribute('src');
      addToCart({ name: productName, price: productPrice, image: productImage })
        .then(success => {
          if (success) showToast('"' + productName + '" added to cart!');

  localStorage.removeItem('svara_user_name');
  localStorage.removeItem('svara_user_id');

  var btn = document.getElementById('profileBtn');
  var dd = document.getElementById('profileDropdown');

    dd.classList.remove('open');
  }
});

<script>
document.getElementById('loginForm').addEventListener('submit', async function(e){
  e.preventDefault();

  const formData = new FormData(this);

  try {
    const res  = await fetch('sql.php', { method:'POST', body: formData });
    const text = await res.text();

    let title, msg, redirect = null;

    if(text.includes('Location: home1.html') || res.redirected){
      window.location.href = 'home1.html';
      return;
    } else if(text.includes('Invalid email or password')){
      title = 'Login Failed';
      msg   = 'Invalid email or password. Please try again.';
    } else if(text.includes('Connection failed')){
      title = 'Connection Error';
      msg   = 'Could not connect to the database. Please try again later.';
    } else if(text.includes('not submitted correctly')){
      title = 'Submission Error';
      msg   = 'The form was not submitted correctly. Please try again.';
    } else {
      title = 'Something went wrong';
      msg   = text.replace(/<[^>]+>/g,'').trim() || 'Please try again.';
    }

    document.getElementById('popupTitle').textContent = title;
    document.getElementById('popupMsg').textContent   = msg;
    document.getElementById('svara-overlay').classList.add('active');

    document.getElementById('popupBtn').onclick = function(){
      document.getElementById('svara-overlay').classList.remove('active');
      if(redirect) window.location.href = redirect;
    };

  } catch(err){
    alert('Network error - please check your connection.');
  }
});
</script>

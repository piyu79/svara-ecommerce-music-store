<script>
// Profile
(function() {
  var name = localStorage.getItem('svara_user_name') || '';
  var btn = document.getElementById('profileBtn');
  var nameEl = document.getElementById('profileName');
  if (name) {
    btn.textContent = name.charAt(0).toUpperCase();
    nameEl.textContent = name;
  }
})();
function toggleProfile() {
  document.getElementById('profileDropdown').classList.toggle('open');
}
function logOut() {
  localStorage.removeItem('svara_user_name');
  localStorage.removeItem('svara_user_id');
  window.location.href = 'landing.html';
}
document.addEventListener('click', function(e) {
  var btn = document.getElementById('profileBtn');
  var dd = document.getElementById('profileDropdown');
  if (btn && dd && !btn.contains(e.target) && !dd.contains(e.target)) {
    dd.classList.remove('open');
  }
});

// Form submit
document.getElementById('updateForm').addEventListener('submit', async function(e) {
  e.preventDefault();
  const formData = new FormData(this);
  formData.append('update_submit', 'true');
  try {
    const res  = await fetch('update_sql.php', { method: 'POST', body: formData });
    const text = await res.text();
    let title, msg, redirect = null;
    if (text.trim().toLowerCase().includes('updated successfully')) {
      title = '✅ Profile Updated!';
      msg   = 'Your information has been updated successfully.';
      // update localStorage name if new name was entered
      const newName = document.getElementById('new_name').value.trim();
      if (newName) localStorage.setItem('svara_user_name', newName);
      redirect = 'home1.html';
    } else if (text.trim().toLowerCase().includes('no account found')) {
      title = 'Email Not Found';
      msg   = 'No account found with that email. Please check and try again.';
    } else if (text.trim().toLowerCase().includes('connection failed')) {
      title = 'Connection Error';
      msg   = 'Could not connect to the database. Please try again later.';
    } else {
      title = 'Something went wrong';
      msg   = text.replace(/<[^>]+>/g, '').trim() || 'Please try again.';
    }
    document.getElementById('popupTitle').textContent = title;
    document.getElementById('popupMsg').textContent   = msg;
    document.getElementById('svara-overlay').classList.add('active');
    document.getElementById('popupBtn').onclick = function() {
      document.getElementById('svara-overlay').classList.remove('active');
      if (redirect) window.location.href = redirect;
    };
  } catch(err) {
    alert('Network error - please check your connection.');
  }
});
</script>

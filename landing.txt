<script>
  function openModal(tab) {
    document.getElementById('authModal').classList.add('active');
    switchTab(tab || 'login');
  }
  function closeModal() {
    document.getElementById('authModal').classList.remove('active');
  }
  function switchTab(tab) {
    document.getElementById('tabLogin').classList.toggle('active', tab === 'login');
    document.getElementById('tabSignup').classList.toggle('active', tab === 'signup');
    document.getElementById('formLogin').classList.toggle('active', tab === 'login');
    document.getElementById('formSignup').classList.toggle('active', tab === 'signup');
  }
  document.getElementById('authModal').addEventListener('click', function(e) {
    if (e.target === this) closeModal();
  });
  function showToast(msg) {
    const t = document.getElementById('toast');
    document.getElementById('toast-msg').textContent = msg;
    t.classList.add('show');
    setTimeout(() => t.classList.remove('show'), 3000);
  }

  // ── LOGIN → saves name → redirect to home1.html ──
  document.getElementById('loginForm').addEventListener('submit', async function(e) {
    e.preventDefault();
    const formData = new FormData(this);
    formData.append('login_submit', '1');
    try {
      const res  = await fetch('sql.php', { method: 'POST', body: formData });
      const text = await res.text();
      if (text.trim().toLowerCase().includes('login successful') || res.redirected) {
        const parts = text.split('|');
        if (parts[1]) localStorage.setItem('svara_user_name', parts[1].trim());
        if (parts[2]) localStorage.setItem('svara_user_id', parts[2].trim());
        closeModal();
        showToast('Login successful! Taking you in...');
        setTimeout(() => { window.location.href = 'home1.html'; }, 1200);
      } else if (text.includes('Invalid email or password')) {
        alert('Invalid email or password. Please try again.');
      } else {
        alert(text.replace(/<[^>]+>/g, '').trim() || 'Something went wrong, please try again.');
      }
    } catch(err) {
      alert('Network error – please check your connection.');
    }
  });

  // ── SIGNUP → redirect to home1.html ──
  document.getElementById('signupForm').addEventListener('submit', async function(e) {
    e.preventDefault();
    const formData = new FormData(this);
    formData.append('register_submit', '1');
    try {
      const res  = await fetch('register_sql.php', { method: 'POST', body: formData });
      const text = await res.text();
      if (text.includes('Registration Successful') || text.includes('Successful')) {
        closeModal();
        showToast('Account created! Please login.');
        setTimeout(() => openModal('login'), 1500);
      } else if (text.includes('already registered')) {
        alert('This email is already registered. Please login instead.');
      } else if (text.includes('do not match')) {
        alert('Passwords do not match!');
      } else {
        alert(text.replace(/<[^>]+>/g, '').trim() || 'Something went wrong.');
      }
    } catch(err) {
      alert('Network error – please check your connection.');
    }
  });
</script>
